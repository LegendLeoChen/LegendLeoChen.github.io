def sample_cfg(model, noise_scheduler, n_samples, in_channels, text_embeddings, image_size=64, guidance_scale=3.0):
    """
    ä»å™ªå£°å¼€å§‹ï¼Œé€æ¸å‡å°å™ªå£°ï¼Œç›´åˆ°æœ€ç»ˆçš„å›¾åƒã€‚

    :param noise_scheduler: å™ªå£°è°ƒåº¦å™¨

    :param in_channels: è¾“å…¥å›¾åƒçš„é€šé“æ•°
    :param text_embeddings: æ–‡æœ¬åµŒå…¥
    :param image_size: å›¾åƒçš„å¤§å°
    :param guidance_scale: ç”¨äºåŠ æƒå™ªå£°é¢„æµ‹çš„æ¯”ä¾‹
    :return: ç”Ÿæˆçš„å›¾åƒ

    device = next(model.parameters()).device

    x = torch.randn(n_samples, in_channels, image_size, image_size).to(device) # éšæœºåˆå§‹åŒ–å™ªå£°å›¾åƒ
    null_embeddings = torch.zeros_like(text_embeddings) # ç”¨äºæ— æ¡ä»¶ç”Ÿæˆ

    # é€æ­¥å»å™ª
    for t in reversed(range(noise_scheduler.num_timesteps)):
        t_batch = torch.full((n_samples,), t, device=device, dtype=torch.long)

        noise_pred_uncond = model(x, t_batch, y=null_embeddings) # ç”Ÿæˆä¸€ä¸ªæ— æ¡ä»¶çš„å™ªå£°é¢„æµ‹
        noise_pred_cond = model(x, t_batch, y=text_embeddings) # ç”Ÿæˆä¸€ä¸ªæœ‰æ¡ä»¶çš„å™ªå£°é¢„æµ‹
        noise_pred = noise_pred_uncond + guidance_scale * (noise_pred_cond - noise_pred_uncond) # CFGï¼šç»“æœåŠ æƒåçš„å™ªå£°é¢„æµ‹

        # é‡‡æ ·å™¨çš„å»å™ªè¿‡ç¨‹

        alpha_t_bar = noise_scheduler.alphas_cumprod[t]

        else:
            noise = torch.zeros_like(x)

        # å»å™ªå…¬å¼

    model.train()
    return x

class StableDiffusion(nn.Module):
    def __init__(self, in_channels=3, latent_dim=4, image_size=512, diffusion_timesteps=1000, device="cuda"):
        super(StableDiffusion, self).__init__()
        # VAE
        self.vae = VAE(in_channels=in_channels, latent_dim=latent_dim, image_size=image_size)
        # Diffusion model (UNet)
        self.unet = UNet_Transformer(in_channels=latent_dim)
        # Noise scheduler
        self.noise_scheduler = NoiseScheduler(num_timesteps=diffusion_timesteps, device=device)

    def encode(self, x):
        return self.vae.encode(x)[0]

    def decode(self, z):
        return self.vae.decode(z)

    def diffuse(self, latents, t, context):
        return self.unet(latents, t, context)

    def forward(self, latents, t, context):
        noise_pred = self.diffuse(latents, t, context)
        return noise_pred
